force_convert_mission = {

	type = neighbor_countries

	category = MIL

	allow = {
		ROOT = { 
			religion = catholic 
			has_country_modifier = counter_reformation
			num_of_cities = 15
			NOT = { has_country_modifier = force_converted_country_mission }
		}
		OR = {
			religion = reformed
			religion = protestant
		}
		NOT = {
			alliance_with = ROOT
			has_opinion = { who = ROOT value = 50 }
			marriage_with = ROOT
		}
		NOT = { num_of_cities = ROOT }
		army_size = ROOT
	}
	abort = {
		OR = {
			ROOT = { NOT = { religion = catholic } }
			AND = {
				NOT = { religion = reformed }
				NOT = { religion = protestant }
			}
		}
	}
	success = {
		religion = ROOT
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 1.1
			ROOT = { has_idea_group = religious_ideas }
		}	
		modifier = {
			factor = 1.25
			is_defender_of_faith = yes
		}
	}
	effect = {
		add_country_modifier = {
			name = "force_converted_country_mission"
			duration = 3650
		}
	}
}


convert_province_mission = {

	type = our_provinces

	category = ADM

	allow = {
		FROM = {
			is_at_war = no
			num_of_missionaries = 1
			NOT = { has_country_flag = converted_province_mission }
		}
		NOT =  { religion = FROM }
		NOT = { has_province_modifier = religious_migration }
	}
	abort = {
		OR = {
			has_province_modifier = religious_migration
			NOT = { owned_by = FROM }
		}
	}
	success = {
		religion = FROM
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 1.05
			FROM = { has_idea_group = religious_ideas }
		}		
		modifier = {
			factor = 1.05
			FROM = {  num_of_missionaries = 2 }
		}
		modifier = {
			factor = 1.05
			FROM = {  is_defender_of_faith = yes }
		}				
	}
	effect = {
		owner = {
			add_country_modifier = {
				name = "force_converted_province_mission"
				duration = 3650
			}
			set_country_flag = converted_province_mission
		}
	}
}

achive_religious_unity = {

	type = country

	category = ADM

	allow = {
		has_factions = no
		NOT = { religious_unity = 0.8 }
	}
	abort = {
	}
	success = {
		religious_unity = 1
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 1.1
			is_defender_of_faith = yes
		}
		modifier = {
			factor = 1.1
			has_idea_group = religious_ideas
		}
		modifier = {
			factor = 1.05
			NOT = { religious_unity = 0.4 }
		}
		modifier = {
			factor = 1.05
			NOT = { religious_unity = 0.5 }
		}
		modifier = {
			factor = 1.05
			NOT = { religious_unity = 0.6 }
		}
		modifier = {
			factor = 1.05
			NOT = { religious_unity = 0.7 }
		}
		modifier = {
			factor = 1.5
			has_idea = devoutness
		}	
	}
	effect = {
		add_country_modifier = {
			name = "achieved_religious_unity_mission"
			duration = 3650
		}
	}
}
