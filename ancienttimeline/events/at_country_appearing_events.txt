namespace = newcntry

country_event = {
	id = newcntry.1
	title = "newcntry.EVTNAME1"
	desc = "newcntry.EVTDESC1"
	picture = CITY_DEVELOPMENT_eventPicture

	fire_only_once = yes

	trigger = {
		NOT = { exists = TOV }
		any_owned_province = { region = britannia_region }
		any_province = {
			region = britannia_region
			is_empty = yes
		}
		is_year = 4700
	}
	mean_time_to_happen = {
		months = 3600
	}
	
	immediate = {
		random_province = {
			limit = {
				region = britannia_region
				is_empty = yes
			}
			cede_province = TOV
			add_core = TOV
			change_trade_goods = cloth
		}
		TOV = { every_neighbor_country = { add_truce_with = TOV } }
	}

	option = {
		name = "newcntry.EVTOPTA1"
	}
}
