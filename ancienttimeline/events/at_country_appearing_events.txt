namespace = newcntry

country_event = {
	id = newcntry.1
	title = "newcntry.EVTNAME1"
	desc = "newcntry.EVTDESC1"
	picture = CITY_DEVELOPMENT_eventPicture

	fire_only_once = yes
	hidden = yes

	trigger = {
		is_year = 4700
		238 = { is_empty = yes }
	}
	mean_time_to_happen = {
		months = 240
	}
	
	immediate = {
		238 = { 
			cede_province = TOV
			add_core = TOV
			change_trade_goods = cloth
		}
		TOV = { every_neighbor_country = { add_truce_with = TOV } }
	}

	option = {
		name = "newcntry.EVTOPTA1"
		
	}
}
